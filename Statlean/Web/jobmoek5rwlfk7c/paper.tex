\begin{theorem}[Hoeffding's theorem]
\label{thm:hoeffding-ustatistic-variance}
Let $X_1,\dots,X_n$ be i.i.d.\ random variables and let
$h(x_1,\dots,x_m)$ be a symmetric Borel kernel with
$E\bigl[h(X_1,\dots,X_m)\bigr]^2 < \infty$. Define the U-statistic
\[
  U_n \;=\; \binom{n}{m}^{-1} \sum_{c} h(X_{i_1},\dots,X_{i_m}),
\]
where the sum $\sum_c$ runs over all $\binom{n}{m}$ subsets
$\{i_1<\dots<i_m\}$ of $\{1,\dots,n\}$. For $k=1,\dots,m$ let
\[
  h_k(x_1,\dots,x_k) \;=\; E\bigl[h(x_1,\dots,x_k,X_{k+1},\dots,X_m)\bigr],
\]
and
\[
  \zeta_k \;=\; \Var\bigl(h_k(X_1,\dots,X_k)\bigr).
\]
Then
\[
  \Var(U_n) \;=\; \binom{n}{m}^{-1} \sum_{k=1}^{m} \binom{m}{k}\binom{n-m}{m-k}\,\zeta_k.
\]
\end{theorem}

\begin{proof}
Set $\tilde h_k = h_k - E[h(X_1,\dots,X_m)]$ and $\tilde h = \tilde h_m$,
so that
\[
  U_n - E(U_n) \;=\; \binom{n}{m}^{-1} \sum_{c} \tilde h(X_{i_1},\dots,X_{i_m}).
\]
Consider two index subsets $\{i_1,\dots,i_m\}$ and $\{j_1,\dots,j_m\}$
of distinct elements of $\{1,\dots,n\}$ having exactly $k$ elements in
common. The number of ordered pairs of such subsets is
$\binom{n}{m}\binom{m}{k}\binom{n-m}{m-k}$. By symmetry of $\tilde h$
and independence of $X_1,\dots,X_n$,
\[
  E\bigl[\tilde h(X_{i_1},\dots,X_{i_m})\,\tilde h(X_{j_1},\dots,X_{j_m})\bigr] \;=\; \zeta_k
\]
for every such pair (this is left as an exercise in Shao). Hence
\[
  \Var(U_n) \;=\; \binom{n}{m}^{-2} \sum_{k=1}^{m} \binom{n}{m}\binom{m}{k}\binom{n-m}{m-k}\,\zeta_k
            \;=\; \binom{n}{m}^{-1} \sum_{k=1}^{m} \binom{m}{k}\binom{n-m}{m-k}\,\zeta_k.
\]
\end{proof}
