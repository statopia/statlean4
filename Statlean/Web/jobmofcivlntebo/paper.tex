\begin{theorem}[Shao 3.5 — Asymptotic distribution of U-statistics]
\label{thm:shao-3-5}
Let $U_n$ be a U-statistic of degree $m$ with kernel $h$, defined by
$U_n = \binom{n}{m}^{-1} \sum_{c} h(X_{i_1}, \ldots, X_{i_m})$
where the sum is over all $\binom{n}{m}$ subsets $\{i_1 < \cdots < i_m\}$ of $\{1,\ldots,n\}$,
and $X_1, X_2, \ldots$ are i.i.d.\ from $P$.  Assume $E[h(X_1, \ldots, X_m)]^2 < \infty$.
Define
\[
  h_k(x_1, \ldots, x_k) = E\bigl[ h(x_1, \ldots, x_k, X_{k+1}, \ldots, X_m) \bigr],
  \qquad
  \zeta_k = \operatorname{Var}\bigl( h_k(X_1, \ldots, X_k) \bigr).
\]
Then:
\begin{enumerate}
\item[(i)] If $\zeta_1 > 0$, then
\[
  \sqrt{n}\bigl[ U_n - E(U_n) \bigr] \;\xrightarrow{d}\; N\bigl( 0,\; m^2 \zeta_1 \bigr).
\]
\item[(ii)] If $\zeta_1 = 0$ but $\zeta_2 > 0$, then
\[
  n\bigl[ U_n - E(U_n) \bigr] \;\xrightarrow{d}\; \frac{m(m-1)}{2} \sum_{j=1}^{\infty} \lambda_j \bigl( \chi^2_{1j} - 1 \bigr),
\]
where $\chi^2_{1j}$ are i.i.d.\ chi-square random variables with one degree of freedom,
and $\lambda_j$ are constants (depending on $P$) satisfying $\sum_{j=1}^{\infty} \lambda_j^2 = \zeta_2$.
\end{enumerate}
\end{theorem}
