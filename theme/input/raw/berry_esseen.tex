\section{Berry--Esseen Bound}

\begin{theorem}[Berry--Esseen Theorem]
There exists a positive universal constant $C$ such that the following holds.
Suppose $Y_1, Y_2, \ldots, Y_n$ are i.i.d.\ random variables with
$E(Y_1) = 0$, $E(Y_1^2) = \sigma^2 > 0$, and $E(|Y_1|^3) = \rho < \infty$.
Let $F_n$ be the CDF of
\[
  \frac{\sum_{i=1}^n Y_i}{\sigma \sqrt{n}}
\]
and $\Phi$ be the CDF of the standard normal distribution. Then
\[
  \sup_{y \in \mathbb{R}} |F_n(y) - \Phi(y)| \le \frac{C \rho}{\sigma^3 \sqrt{n}}.
\]
\end{theorem}

\begin{proof}
The proof uses characteristic functions (Fourier transforms).

\textbf{Step 1 (Smoothing inequality).}
By the smoothing lemma (Berry's inequality), for any $T > 0$,
\[
  \sup_y |F_n(y) - \Phi(y)|
  \le \frac{1}{\pi} \int_{-T}^{T} \left| \frac{\varphi_n(t) - e^{-t^2/2}}{t} \right| dt
  + \frac{C_1}{T},
\]
where $\varphi_n(t) = [\varphi(t/(\sigma\sqrt{n}))]^n$ is the characteristic function of
$\sum_i Y_i / (\sigma\sqrt{n})$ and $\varphi$ is the characteristic function of $Y_1$.

\textbf{Step 2 (Taylor expansion of characteristic function).}
Since $E Y_1 = 0$ and $E Y_1^2 = \sigma^2$, we have for $|t| \le \sigma\sqrt{n}$:
\[
  \varphi(t/(\sigma\sqrt{n})) = 1 - \frac{t^2}{2n} + r_n(t),
  \quad |r_n(t)| \le \frac{\rho |t|^3}{6 \sigma^3 n^{3/2}}.
\]

\textbf{Step 3 (Comparison with Gaussian).}
Using $|a^n - b^n| \le n|a-b| \max(|a|,|b|)^{n-1}$ and the bound
$|1 - t^2/(2n)|^n \to e^{-t^2/2}$, one shows
\[
  |\varphi_n(t) - e^{-t^2/2}| \le \frac{C_2 \rho |t|^3}{\sigma^3 \sqrt{n}} e^{-t^2/3}
\]
for $|t| \le \sigma\sqrt{n}/\rho$ (the ``small $t$'' regime).

\textbf{Step 4 (Choosing $T$ and combining).}
Take $T = \sigma^3 \sqrt{n} / \rho$. The integral over $|t| \le T$ contributes
$O(\rho/(\sigma^3\sqrt{n}))$, and the boundary term $C_1/T$ also contributes
$O(\rho/(\sigma^3\sqrt{n}))$.
\end{proof}
