\begin{theorem}[Bayes' rule]
Let $(\Omega, \mathcal{F}, \mathbb{P})$ be a probability space and $A, B \in \mathcal{F}$ events with
$\mathbb{P}(B) > 0$. Then
\[
\mathbb{P}(A \mid B) = \frac{\mathbb{P}(B \mid A)\, \mathbb{P}(A)}{\mathbb{P}(B)}.
\]
\end{theorem}

\begin{proof}
By definition of conditional probability,
$\mathbb{P}(A \mid B) = \mathbb{P}(A \cap B) / \mathbb{P}(B)$
and
$\mathbb{P}(B \mid A)\, \mathbb{P}(A) = \mathbb{P}(A \cap B)$.
Substituting gives the claim.
\end{proof}
